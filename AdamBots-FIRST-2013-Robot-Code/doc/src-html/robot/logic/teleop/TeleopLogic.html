090            
091            // If the left and right drive variables are both equal to 0 and the auto target button is held then let's auto target
092            if (_leftDrive == 0 && _rightDrive == 0 && _primaryButtons[FancyJoystick.BUTTON_RB]) {
093                //TODO: Implement auto targeting, values will be assigned to right and left drive here.
094            }
095            
096            // Keeps the drive variables in their -1 to 1 range
097            _leftDrive = Math.max(-1, Math.min(1, _leftDrive));
098            _rightDrive = Math.max(-1, Math.min(1, _rightDrive));
099            
100            _robotDrive.drive(_leftDrive, _rightDrive);
101            
102            // Handle shifting
103            if (_primaryButtons[FancyJoystick.BUTTON_LB]) {
104                if (_primaryButtons[FancyJoystick.BUTTON_A]) {
105                    _highGear=false;
106                } else if (_primaryButtons[FancyJoystick.BUTTON_Y]) {
107                    _highGear=true;
108                }
109            }
110            
111            // Infeed roller for pickup mechanism
112            if (_primaryButtons[FancyJoystick.BUTTON_X]) {
113                //TODO: Run pickup roller
114            } else if (_primaryButtons[FancyJoystick.BUTTON_B]) {
115                //TODO: Reverse pickup roller
116            } else {
117                //TODO: Pickup roller off
118            }
119            
120            //// SECONDARY DRIVER --------------------------------------------------
121            _shooterAngleChangerDrive = _secondaryAxis[FancyJoystick.AXIS_LEFT_Y];
122            
123            if (_secondaryButtons[FancyJoystick.BUTTON_RB]) {
124                if (_shooterAngleChangerDrive == 0 && _magicBoxButtons[MagicBox.AUTO_ANGLE_ENABLED]) {
125                    //TODO: Add automatic shooter angle adjustment
126                }
127                
128                if (_magicBoxButtons[MagicBox.AUTO_SHOOTER_SPEED_ENABLED]) {
129                    //TODO: Add automatic shooter speed calls
130                }
131            }
132            
133            //TODO: Run shooter angle changer
134            
135            // Drive elevator
136            _elevatorDrive = _secondaryAxis[FancyJoystick.AXIS_TRIGGERS];
137            //TODO: Drive elevator
138            
139            // Disk fire control
140            if (_secondaryButtons[FancyJoystick.BUTTON_A]) {
141                //TODO: Extend firing solenoid
142            } else {
143                //TODO: Retract firing solenoid
144            }
145        }
146        
147        /**
148         * Gathers joystick button and axis values and stores them in their respective arrays.
149         */
150        private void updateJoystickValues() {
151            for (int i = 1; i < FancyJoystick.XBOX_BUTTONS; i++) {
152                _primaryButtons[i] = _primaryJoy.getRawButton(i);
153                _secondaryButtons[i] = _secondaryJoy.getRawButton(i);
154            }
155            
156            for (int i = 0; i < FancyJoystick.XBOX_AXES; i++) {
157                _primaryAxis[i] = _primaryJoy.getDeadAxis(i);
158                _secondaryAxis[i] = _secondaryJoy.getDeadAxis(i);
159            }
160        }
161        
162        /**
163         * Gathers magic box values.
164         */
165        private void updateMagicBox() {
166            for (int i = 0; i < MagicBox.NUM_BUTTONS; i++) {
167                _magicBoxButtons[i] = _magicBox.getDigitalIn(i);
168            }
169        }
