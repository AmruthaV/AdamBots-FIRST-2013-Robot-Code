005    package robot.logic;
006    
007    import edu.wpi.first.wpilibj.PIDController;
008    import edu.wpi.first.wpilibj.Timer;
009    import edu.wpi.first.wpilibj.Victor;
010    import robot.sensors.FancyCounter;
011    
012    /**
013     *
014     * @author Tyler
015     */
016    public class PIDLogic {
017        //// VARIABLES -------------------------------------------------------------
018        // Victor instances
019        private Victor _driveMotors;
020        
021        // PID constants
022        private double _Ki = 0.0;
023        private double _Kp = 0.0;
024        private double _Kd = 0.0;
025        
026        public final double PID_TOLERANCE = 5.0;
027        public final double MAX_INPUT = 10000;
028        public final double MIN_INPUT = 0;
029        public final double MAX_OUTPUT = 1.0;
030        public final double MIN_OUTPUT = -1.0;
031        public final int    PID_BUFFER = 5;
032        
033        // Buffer variables
034        private int _targetBuffer;
035        
036        // Timer Variables
037        //private Timer _timer = new Timer();
038        
039        // Port Constants
040        //public final int COUNTER_PORT = 1;
041        
042        // Counter instances
043        private FancyCounter _driveCounter;
044        
045        // PIDController instances
046        private PIDController _pid;
047        
048        //// CONSTRUCTOR -----------------------------------------------------------
049        public PIDLogic(Victor driveMotor, FancyCounter driveCounter, double Ki, double Kp, double Kd) {
050            pidsInit();
051            _Ki = Ki;
052            _Kp = Kp;
053            _Kd = Kd;
054            _driveMotors = driveMotor;
055            _driveCounter = driveCounter;
056            _targetBuffer = 0;
057        }
058        
059        //// INIT METHOD -----------------------------------------------------------
060        public void pidsInit() {
061            //// INITIALIZING ------------------------------------------------------
062            //_driveCounter = new FancyCounter(COUNTER_PORT);
063            _pid = new PIDController(_Kp, _Ki, _Kd, _driveCounter, _driveMotors);
064            
065            //// STARTING ----------------------------------------------------------
066            _driveCounter.start();
067            _driveCounter.setMaxPeriod(100000);
068            _driveCounter.setUpSourceEdge(true, true);
069            _pid.setInputRange(MIN_INPUT, MAX_INPUT);
070            _pid.setOutputRange(MIN_OUTPUT, MAX_OUTPUT);
071            _pid.setPercentTolerance(PID_TOLERANCE);
072            _pid.enable();
073            //_timer.start();
074        }
075        
076        //// PID METHODS -----------------------------------------------------------
077        // sets the RPM
078        public void setRPM(double targetRPM) {
079            System.out.println("SETTING RPM");
080            
081            // enables the PID if it isn;t enabled
082            if (!_pid.isEnable()) {
083                _pid.enable();
084            }
085            
086            // sets the set point
087            if (_pid.getSetpoint() != targetRPM) {
088                _pid.setSetpoint(targetRPM);
089            }
090            //System.out.println(timer.get() + "\t" + topPid.get() + "\t" + _topGear.pidGet() + "\t" + bottomPid.get() + "\t" + _bottomGear.pidGet() + "\t" + shooterToSpeed());
091        }
092        
093        // checks if it's to speed
094        public boolean toSpeed() {
095            // checks if the PID is on target
096            if (_pid.onTarget()) {
097                if (_targetBuffer <= PID_BUFFER) {
098                    _targetBuffer++;
099                }
100            } else {
101                _targetBuffer = 0;
102            }
103            
104            // checks how long the _pid had been on target
105            if (_targetBuffer > PID_BUFFER) {
106                return true;
107            } else {
108                return false;
109            }
110        }
111    }
