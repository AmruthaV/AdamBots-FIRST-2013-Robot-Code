007    /**
008     *
009     * @author Tyler
010     */
011    public class FancyPIDController extends PIDController {
012        
013        // Buffer variables, keep track of # of iterations on target
014        private int _targetBuffer = 0;
015        
016        // # of iterations of good readings where target is considered met
017        public final int    PID_BUFFER = 5;
018        
019    
020        //// CONSTRUCTOR -----------------------------------------------------------
021        public FancyPIDController( double Kp, double Ki, double Kd, PIDSource source, PIDOutput output ) {
022            super( Kp, Ki, Kd, source, output );     
023        }
024        
025        //// PID METHODS -----------------------------------------------------------
026        // sets the RPM
027        public void setRPM(double targetRPM) {
028            System.out.println("SETTING RPM");
029            
030            // enables the PID if it isn't enabled
031            if (!super.isEnable()) {
032                super.enable();
033            }
034            
035            // sets the set point (Micklas - need to look at PIDController 
036            // source, even though this looks unnecessary I think we needed it
037            // because setSetpoint has some additional logic that resets
038            // accumulations?  or something like that
039            if (super.getSetpoint() != targetRPM) {
040                super.setSetpoint(targetRPM);
041            }
042           // debug in case we need to tune PIDs
043           // System.out.println(timer.get() + "\t" + topPid.get() + "\t" + _topGear.pidGet() + "\t" + bottomPid.get() + "\t" + _bottomGear.pidGet() + "\t" + shooterToSpeed());
044        }
045        
046        // checks if it's to speed
047        public boolean isAtSpeed() {
048            // checks if the PID is on target
049            if (super.onTarget()) {
050                if (_targetBuffer <= PID_BUFFER) {
051                    _targetBuffer++;
052                }
053            } else {
054                _targetBuffer = 0;
055            }
056            
057            // checks how long the _pid had been on target
058            if (_targetBuffer > PID_BUFFER) {
059                return true;
060            } else {
061                return false;
062            }
063        }
064    }
