005    package robot.logic;
006    
007    /**
008     * Contains methods that should be implemented by each "logic phase".  Each of the three logic phases extends this class (AutonLogic, ClimbLogic, TeleopLogic).
009     * @author Ben
010     * @see auton.AutonLogic
011     * @see climb.ClimbLogic
012     * @see TeleopLogic
013     */
014    public interface ILogicPhase {
015        
016        /**
017         * Called when this Logic Phase is granted control of the robot.
018         */
019        public void init();
020        
021        /**
022         * Called periodically while this Logic Phase possesses control.
023         */
024        public void update();
025        
026        /**
027         * Called immediately before control is taken from this Logic Phase.
028         */
029        public void finish();
030    }
