005    package robot.logic;
006    
007    /**
008     *
009     * @author Ben
010     */
011    public abstract class LogicTask {
012        
013        /** Success Status */
014        public static final int SUCCESS = 0;
015        /** Failure Status */
016        public static final int FAILURE = 0;
017        
018        //// CONSTRUCTOR -----------------------------------------------------------
019        
020        public LogicTask(){
021            
022        }
023        
024        //// INITIALIZATION --------------------------------------------------------
025        
026        /**
027         * Called when the Task is started.
028         */
029        public abstract void initializeTask();
030        
031        /**
032         * Called periodically while the Task is being executed.
033         */
034        public abstract void updateTask();
035        
036        /**
037         * Called when the Task is stopped or completed.
038         * @return A status message.  (0 = success)
039         */
040        public abstract int finishTask();
041        
042        /**
043         * Indicates whether or not the task has been completed.
044         * @return Has the task been completed?
045         */
046        public abstract boolean isDone();
047    }
