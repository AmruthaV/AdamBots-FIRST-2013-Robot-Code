005    package robot.logic.tasks;
006    
007    import robot.behavior.RobotClimb;
008    import robot.behavior.RobotShoot;
009    import robot.logic.LogicTask;
010    import robot.sensors.RobotSensors;
011    
012    /**
013     *
014     * @author Ben
015     */
016    public final class TAwaitStatus extends LogicTask {
017        //// CONSTANTS -------------------------------------------------------------
018        
019        public static final int WINCH_IN_POSITION = 0;
020        public static final int SHOOTER_IN_POSITION = 1;
021        public static final int SHOOTER_UP_TO_SPEED = 2;
022        public static final int WINCH_HOOK_LIMIT_SWITCHES_PRESSED = 3;
023        
024        //// PRIVATE VARIABLES -----------------------------------------------------
025        
026        private int _status;
027        private double _value;
028        
029        private boolean _done = false;
030        
031        //// CONSTRUCTOR -----------------------------------------------------------
032        
033        public TAwaitStatus(int status){
034            this(status, 4.0);
035        }
036        
037        public TAwaitStatus(int status, double value){
038            _status = status;
039            _value = value;
040            initializeTask();
041        }
042    
043        //// INITIALIZATION --------------------------------------------------------
044        
045        /**
046         * Performs any necessary initialization for this task and optionally sends
047         * an expected value to a behavior class.  For example, if this task is set
048         * to wait for the WINCH_IN_POSITION status, initializeTask() will call
049         * RobotClimb.setWinchTarget() with the value given to this task's
050         * constructor.
051         * @param useValue Should the Task send a target value to the Behavior class
052         * that it is waiting on?
053         */
054        public void initializeTask() {
055            if(_value != Double.NaN){
056                switch(_status){
057                    case WINCH_IN_POSITION:
058                        RobotClimb.setWinchTarget(_value);
059                        break;
060                    case SHOOTER_IN_POSITION:
061                        RobotShoot.setAngleDegrees(_value);
062                        break;
063                    case SHOOTER_UP_TO_SPEED:
064                        RobotShoot.setSpeed(_value);
065                        break;
066                    case WINCH_HOOK_LIMIT_SWITCHES_PRESSED:
067                        break;
068                    default:
069                            throw new IllegalArgumentException();
070                }
071            }
072        }
073    
074        //// UPDATE ----------------------------------------------------------------
075        
076        public void updateTask() {
077            switch(_status){
078                case WINCH_IN_POSITION:
079                    _done = RobotClimb.isWinchInPosition();
080                    break;
081                case SHOOTER_IN_POSITION:
082                    _done = RobotShoot.isShooterInPosition();
083                    break;
084                case SHOOTER_UP_TO_SPEED:
085                    _done = false; // RobotShoot.isShooterUpToSpeed(); 
086                    // TODO:  Shooter Up to Speed Method
087                    break;
088                case WINCH_HOOK_LIMIT_SWITCHES_PRESSED:
089                    _done = RobotSensors.limitHookLeftArm.get() && RobotSensors.limitHookRightArm.get();
090                    break;
091            }
092        }
093    
094        //// FINISH ----------------------------------------------------------------
095        
096        /**
097         * Stops this task, regardless of completion status.  Returns a status value
098         * indicating success or failure.
099         * @return Status value.  (SUCCES=0, FAILURE=1)
100         * @see robot.logic.LogicTask#SUCCESS
101         * @see robot.logic.LogicTask#FAILURE
102         */
103        public int finishTask() {
104            return isDone() ? SUCCESS : FAILURE;
105        }
106        
107        //// COMPLETION STATUS -----------------------------------------------------
108        
109        /**
110         * Has the desired status been reached (thus completing the Task)?
111         * @return Task completion status.
112         */
113        public boolean isDone(){
114            return _done;
115        }
116        
117    }
