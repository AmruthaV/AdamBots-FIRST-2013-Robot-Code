005    package robot.logic;
006    
007    /**
008     * Contains methods that should be implemented by each "logic phase".  
009     * Each of the three logic phases extends this class (AutonLogic, ClimbLogic, TeleopLogic).
010     * @author Ben
011     * @see auton.AutonLogic
012     * @see climb.ClimbLogic
013     * @see TeleopLogic
014     */
015    public abstract class LogicPhase {
016        
017        /** Autonomous Phase Constant. */
018        public static final int AUTONOMOUS = 0;
019        /** Teleop Phase Constant. */
020        public static final int TELEOP = 1;
021        /** Climb Phase Constant. */
022        public static final int CLIMB = 2;
023        
024        /**
025         * Contains methods that should be implemented by each "logic phase".
026         * Each of the three logic phases extends this class (AutonLogic, ClimbLogic, TeleopLogic).
027         */
028        public LogicPhase(){
029            
030        }
031        
032        /**
033         * Called when this Logic Phase is granted control of the robot.
034         */
035        public abstract void initPhase();
036        
037        /**
038         * Called periodically while this Logic Phase possesses control.
039         */
040        public abstract void updatePhase();
041        
042        /**
043         * Called immediately before control is taken from this Logic Phase.
044         */
045        public abstract void finishPhase();
046        
047    }
