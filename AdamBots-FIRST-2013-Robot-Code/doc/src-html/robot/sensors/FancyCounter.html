005    package robot.sensors;
006    
007    import edu.wpi.first.wpilibj.Counter;
008    import edu.wpi.first.wpilibj.PIDSource;
009    
010    /**
011     *
012     * @author Adambots 245
013     */
014    public class FancyCounter extends Counter implements PIDSource{
015        double lastRpm;
016        int errorVal;
017        
018        public FancyCounter(int channel){
019            super(channel);
020            lastRpm = 0;
021            errorVal = 0;
022        }
023        
024        public int getError () {
025            
026            return errorVal;
027        }
028        
029        public void resetError () {
030            errorVal = 0;
031        }
032        
033        public double pidGet () {
034            // double time = getPeriod()
035            double time = getPeriod() * 2;       // might be wrong, not sure if the * 2 is needed
036            double rpm = 60 / time;
037            
038            // if (rom > 5000)  // was that before, but i thought i might need something different that wasn't 5000
039            if (rpm > 10000) {
040                errorVal++;
041                rpm = lastRpm;
042            }
043            
044            lastRpm = rpm;
045            
046            return rpm;
047        }
048    }
