128            /**
129             * Internal image-processing which isolates the green board.
130             */
131            private static void greenBox() {
132                    MonoImage saturationHSVOriginal = null;
133                    MonoImage hueHSVOriginal = null;
134                    MonoImage valueOriginal = null;
135    
136                    BinaryImage result = null;
137                    try {
138                            saturationHSVOriginal = _srcImage.getHSVSaturationPlane();
139                            valueOriginal = _srcImage.getValuePlane();
140                            hueHSVOriginal = _srcImage.getHSVHuePlane();
141                            saturationHSVOriginal = _srcImage.getHSVSaturationPlane();
142    
143                            _srcImage.replaceRedPlane(hueHSVOriginal);
144                            _srcImage.replaceGreenPlane(saturationHSVOriginal);
145                            _srcImage.replaceBluePlane(valueOriginal);
146    
147                            result = _srcImage.thresholdRGB(110, 140, 200, 256, 225, 256);
148    
149                            ParticleAnalysisReport[] greens = result.getOrderedParticleAnalysisReports();
150                            ParticleAnalysisReport board = null;
151                            for (int i = 0; i < greens.length; i++) {
152                                    if ( (board == null || board.particleArea < greens[i].particleArea) && greens[i].particleArea < greens[i].boundingRectWidth * greens[i].boundingRectHeight * 0.55 ) {
153                                            board = greens[i];
154                                    }
155                            }
156                            double size = board.particleArea;
157                            ParticleAnalysisReport q = null;
158                            for (int i = 0; i < greens.length; i++) {
159                                    if (greens[i].particleArea> size * 0.5)
160                                    {
161                                            System.out.println("Candidate: " + greens[i].boundingRectLeft + "," + greens[i].boundingRectTop);
162                                    }
163                                    if ( (greens[i].particleArea > size * 0.5 && greens[i].particleArea < greens[i].boundingRectWidth * greens[i].boundingRectHeight * 0.55) && (q == null || q.center_mass_y > greens[i].center_mass_y) ) {
164                                            q = greens[i];
165                                    }
166                            }
167                            board = q;
168                            _greenTarget = new Target(board.boundingRectLeft, board.boundingRectTop, board.boundingRectWidth, board.boundingRectHeight);
169                            System.out.println("Target location:" + _greenTarget.x + " ," + _greenTarget.y + " |w,h,a " + _greenTarget.w + "|" + _greenTarget.h + " ," + board.particleArea);
170                    }
171                    catch (NIVisionException e) {
172                    }
173                    finally {
174                            try {
175                                    free(valueOriginal);
176                                    free(saturationHSVOriginal);
177                                    free(hueHSVOriginal);
178                                    free(result);
179                            }
180                            catch (NIVisionException e) {
181                            }
182                    }
183            }

185            /**
186             Initializes AxisCamera instance and sets camera parameters. Should be called once, at robot initialization.
187             **/
188            public static void init() {
189                    //how it will be on the robot ; _camera = AxisCamera.getInstance("10.2.45.11");  // get an instance ofthe camera
190                    _camera = AxisCamera.getInstance("192.168.0.90");
191                    _camera.writeMaxFPS(10);
192                    _camera.writeExposurePriority(AxisCamera.ExposurePriorityT.frameRate);
193                    _camera.writeResolution(AxisCamera.ResolutionT.k320x240);
194                    _camera.writeCompression(10);
195            }

212            /**
213             * Performs "work" on the image, excluding "greenbox." Called by CameraThread only.
214             1. Grabs source image.
215             2. Initiates blueBox() or redBox()
216             3. Free all objects
217             4. Initiates whiteBox()
218             5. Free all objects
219             6. Initiate postImageAnalysisAnalysis(), establishing a _whiteTarget object.
220             7. Write to distance, theta using _whiteTarget.
221             **/
222            public static void work() {
223                    _srcImage = null;
224                    try // Lots of exceptions can happen
225                    {
226                            _srcImage = _camera.getImage();
227                            _srcImage.write("/raw.png");
228                            greenBox();//Depending on team color...
229                            //Need to determine how switch works.
230                            _recentDistanceInches = 14874.0 / ((_greenTarget.w + _greenTarget.h) / 2.0);
231                            //_recentDistanceInches = (TARGET_HEIGHT_INCHES / 2) / Math.tan(_greenTarget.hprime * Math.toRadians(VIEW_ANGLE_DEGREES*VIEW_HEIGHT_OVER_WIDTH) / (VIEW_ANGLE_PIXELS*VIEW_HEIGHT_OVER_WIDTH) / 2.0);
232                            //_recentDistanceInches *= _greenTarget.h / 24;
233                            _recentThetaDegrees = (double) (_greenTarget.x + _greenTarget.w / 2 - VIEW_ANGLE_PIXELS / 2.0) * (VIEW_ANGLE_DEGREES) / (VIEW_ANGLE_PIXELS);
234                            _freshImage = true;
235                    }
236                    catch (Exception e) {
237                    }
238                    finally {
239                            try {
240                                    free(_srcImage);
241                            }
242                            catch (NIVisionException e) {
243                            }
244                    }
245            }
