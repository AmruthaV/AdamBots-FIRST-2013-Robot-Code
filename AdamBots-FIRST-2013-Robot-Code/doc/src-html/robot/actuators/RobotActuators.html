001    package robot.actuators;
002    
003    import edu.wpi.first.wpilibj.Relay;
004    import edu.wpi.first.wpilibj.Servo;
005    import edu.wpi.first.wpilibj.Talon;
006    import edu.wpi.first.wpilibj.Victor;
007    
008    /**
009     * Contains static instances of robot actuators.
010     * @author Ben
011     */
012    public class RobotActuators {
013        //// ENFORCE NONINSTANTIABILITY --------------------------------------------
014        
015        private RobotActuators() throws Exception {
016            throw new java.lang.Exception("You can't instantiate RobotActuators, silly!");
017        }
018        
019        //// INITIALIZATION --------------------------------------------------------
020        
021        /**
022         * Initializes static actuator instances.
023         */
024        public static void init(){
025            driveLeft           = new Victor(1);
026            driveRight          = new Victor(2);
027            shooterWheelMotor   = new Victor(3);
028            shooterAngleMotor   = new Talon(4);
029            discElevator        = new Talon(5);
030            climbWinch          = new Talon(6);
031            transmissionLeft    = new Servo(7);
032            transmissionRight   = new Servo(8);
033            
034            discIntake          = new Relay(1);
035            shooterFeederSolenoid = new Relay(2);
036            hopperSolenoid = new Relay(3);
037            climbWinchSolenoid   = new Relay(4);
038            
039            discIntake.setDirection(Relay.Direction.kBoth);
040            shooterFeederSolenoid.setDirection(Relay.Direction.kForward);
041            hopperSolenoid.setDirection(Relay.Direction.kBoth);
042            climbWinchSolenoid.setDirection(Relay.Direction.kForward);
043        }
044        
045        //// DRIVE -----------------------------------------------------------------
046        
047        /** Left drive Victor. */
048        public static Victor driveLeft;
049        /** Right drive Victor. */
050        public static Victor driveRight;
051        
052        /** Left transmission Servo.*/
053        public static Servo transmissionLeft;
054        /** Right transmission Servo. */
055        public static Servo transmissionRight;
056        
057        //// CLIMBING --------------------------------------------------------------
058        
059        /** Climbing Winch. */
060        public static Talon climbWinch;
061        /** Winch ratchet release solenoid. */
062        public static Relay climbWinchSolenoid;
063        
064        //// DISC ACQUISITION ------------------------------------------------------
065        
066        /** Disc pickup elevator winch. */
067        public static Talon discElevator;
068        /** Spike to control the intake roller. */
069        public static Relay discIntake;
070        /** Controls the hopper solenoid. */
071        public static Relay hopperSolenoid;
072        
073        //// SHOOTER ---------------------------------------------------------------
074        
075        /** Shooter wheel motor. */
076        public static Victor shooterWheelMotor;
077        /** Controls the angle of attack of the shooter. */
078        public static Talon shooterAngleMotor;
079        /** Shooter feeder solenoid. */
080        public static Relay shooterFeederSolenoid;
081    }
